\newglossaryentry{argument}{
  name={$\dottedbox$},
  sort={argument},
  description={arbitrary argument or mathematical object}
}
\newglossaryentry{condition}{
  name={$\condition{\dottedbox}$},
  sort={condition},
  description={symbolic representation of a conditional test: returns 1 if the condition in square brackets is satisfied, else 0 (i.e. generalisation of a characteristic function of the argument in square brackets)}
}
\newglossaryentry{cardinality}{
  name={$\abs{\dottedbox}$},
  sort={cardinality},
  description={absolute value or, if the argument is a set, cardinality}
}
